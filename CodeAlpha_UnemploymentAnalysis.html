# Step 3: Rename columns for clarity (Region  State)
df.rename(columns={'Region': 'State'}, inplace=True)

# Step 4: Clean column names  remove spaces and replace with underscores
df.columns = [col.strip().replace(" ", "_") for col in df.columns]

# Step 5: Convert the Date column to datetime format
df['Date'] = pd.to_datetime(df['Date'], dayfirst=True)

# Step 6: Calculate average unemployment rate by state
state_avg = df.groupby('State')['Estimated_Unemployment_Rate_(%)'].mean().sort_values(ascending=False)

#  Step 7: Bar plot  Average unemployment rate by state
plt.figure(figsize=(12, 6))
state_avg.plot(kind='bar', color='teal')
plt.title("Average Unemployment Rate by State")
plt.xlabel("State")
plt.ylabel("Unemployment Rate (%)")
plt.xticks(rotation=45)
plt.grid(True)
plt.tight_layout()
plt.show()
In[11]:
#  Step 8: Line plot  Unemployment rate over time for each state
plt.figure(figsize=(14, 6))
sns.lineplot(data=df, x='Date', y='Estimated_Unemployment_Rate_(%)', hue='State')
plt.title("Unemployment Rate Over Time by State")
plt.xlabel("Date")
plt.ylabel("Unemployment Rate (%)")
plt.legend(loc='upper right', bbox_to_anchor=(1.2, 1))
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

#  Step 9: Histogram  Distribution of unemployment rates overall
